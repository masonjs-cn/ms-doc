<template>
  <div class="demo-base">
    <ms-table :data="data" :option="option" v-model:page="page"></ms-table>
  </div>

<script setup lang="ts">
import { reactive, ref, h } from "vue";
import { MsButtonGroup } from "@masonjs/components";
import { ElButton } from "element-plus";
const page = ref({
  total: 5,
  currentPage: 1,
  pageSize: 5,
});

const data = Array.from({ length: 5 }).map((_, index) => ({
  key: index,
  name: `Edward King ${index}`,
  age: 32,
  address: `London, Park Lane no. ${index}`,
}));

const btnOptions = reactive<any>([
  {
    label: "编辑",
    link: true,
    key: "edit",
  },
  {
    label: "删除",
    key: "del",
    type: "danger",
  },
  {
    label: "权限",
    key: "edit2",
    hide: true,
  },
  {
    label: "插槽",
    key: "edit2",
    slots: true,
    render() {
      return h(
        ElButton,
        {
          type: "info",
          quaternary: true,
          size: "small",
          onClick: () => {
            console.log(2);
          },
        },
        { default: () => "插槽" }
      );
    },
  },
]);

const createColumns = ({ play }: { play: (row: Song) => void }): DataTableColumns<Song> => {
  return [
    {
      title: "key",
      key: "key",
    },
    {
      title: "name",
      key: "name",
    },
    {
      title: "age",
      key: "age",
    },
    {
      title: "address",
      key: "address",
    },
    {
      title: "Action",
      key: "actions",
      render(row) {
        return h(MsButtonGroup, {
          option: btnOptions,
          onUpdateEvent: (key: string) => {
            console.log(key);
          },
        });
      },
    },
  ];
};

const option = reactive({
  rowKey: "key",
  serial: true,
  columns: createColumns({
    play(row: Song) {
      console.log(row);
    },
  }),
});
</script>

API ​

Table Props ​

名称类型说明默认值
pageObject<IPage> | false分页false
optionObject<ITableOption>见下

Page Props ​

名称类型说明默认值
totalnumber总数总数
currentPagenumber当前页当前页
pageSizenumber每页条数当前页面的总条数

Options Props ​

名称类型说明默认值
rowKey(rowData: object) => (number/string)
serialboolean序号通过行数据创建行的 key（如果你不想给每一行加上 key）
columnsArray<DataTableColumn>栏目[]
selectcheckbox | radio | undefined多选、单选undefined
sizeboolean大小按钮false
fullScreenboolean全屏按钮false
columnsSettingboolean栏目选择false
columnsSortboolean排序false

Methods Props ​

名称类型说明
onLoad触发加载触发、分页加载
update:pagePage返回 page 分页信息
update:checkBoxids[]返回选择的数据
